project(buff_interfaces)

  add_compile_options(-Wall -Wextra -Wpedantic)
endif()

# find dependencies

  "msg/Blade.msg"
  "msg/BladeArray.msg"
  "msg/DebugBlade.msg"
  "msg/DebugBladeArray.msg"
  "msg/Rune.msg"
  "msg/RuneInfo.msg"
  "msg/TimeInfo.msg"

  DEPENDENCIES
    std_msgs
    geometry_msgs
)


ament_package()
